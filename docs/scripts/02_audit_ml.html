Detección de Anomalías con Isolation Forest

import pandas as pd
import numpy as np
from sklearn.ensemble import IsolationForest
from sklearn.preprocessing import StandardScaler

# Obtener datos de R
r_data = r.master_data

# Convertir a DataFrame de pandas
df = pd.DataFrame({
    'temperatura': r_data['temperatura'],
    'plancton': r_data['plancton'],
    'nitratos': r_data['nitratos']
})

# Normalizar datos para mejor detección
scaler = StandardScaler()
df_scaled = scaler.fit_transform(df)

# Aplicar Isolation Forest
# contamination: proporción esperada de outliers (ajustar según necesidad)
iso_forest = IsolationForest(
    contamination=0.01,  # ~1% de outliers esperados
    random_state=42,
    n_estimators=100
)

anomaly_labels = iso_forest.fit_predict(df_scaled)
anomaly_scores = iso_forest.score_samples(df_scaled)

# Crear DataFrame con resultados y pasar a R
results_df = pd.DataFrame({
    'anomaly_label': anomaly_labels,  # -1 = anomalía, 1 = normal
    'anomaly_score': anomaly_scores
})

# Pasar resultados a R de forma explícita
r.anomaly_labels = anomaly_labels
r.anomaly_scores = anomaly_scores
